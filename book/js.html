倒计时

<h1></h1>
<script type="text/javascript">
        function fun(){
        var target = new Date("2017-3-30 14:00");
        var now=new Date();
        var target1 = target.getTime();
        var now1 = now.getTime();
        var tTime=target1-now1;
        var day = Math.floor(tTime / (1000*60*60*24));
        var subday = Math.floor(tTime % (1000*60*60*24));

        var hours = Math.floor(subday / (1000*60*60));
        var subhours = Math.floor(subday % (1000*60*60));

        var  minutes= Math.floor(subhours / (1000*60));
        var  subminutes= Math.floor(subhours % (1000*60));

        var  seconds= Math.floor(subminutes /(1000));
        var allTime=day+':'+hours+':'+minutes+':'+seconds;
        document.querySelector('h1').innerHTML=allTime;

        }
        setInterval(fun,1000)
    </script>
</body>
